#include <stdio.h>
#include <math.h>
int main()
{
    int N;
    char c;

    scanf("%d %c", &N, &c);
    int M = round(N/2.0);
    printf("%d\n", M);
    for(int i = 0; i< M;i++)
    {
        if(i == 0|| i == M-1){
            for(int j = 0; j< N;j++)
            {
                printf("%c", c);
            }
            printf("\n");
        }else{
            printf("%c", c);
            for(int j = 1; j <N-1;j++)
            {
                printf(" ");
            }
            printf("%c\n", c);
        }

    }

    return 0;
}
